121     def breadcrums
 
122       # breadcrums use the request.uri and the instance vars of the parent
 
123       # resources (@branch, @center) that are available -- so no db queries
 
124       crums, url = [], ''
 
125       request.uri.split("?")[0][1..-1].split('/').each_with_index do |part, index|
 
126         url  << '/' + part
 
127         if part.to_i.to_s.length == part.length  # true when a number (id)
 
128           o = instance_variable_get('@'+url.split('/')[-2].singular)  # get the object (@branch)
 
129           s = (o.respond_to?(:name) ? link_to(o.name, url) : link_to('#'+o.object_id.to_s, url))
 
130           crums[-1] += ": <b><i>#{s}</i></b>"  # merge the instance names (or numbers)
 
131         else  # when not a number (id)
 
132           crums << link_to(part.gsub('_', ' '), url)  # add the resource name
 
133         end
 
134       end
 
135       ['', crums].join('&nbsp;<b>&gt;&gt;</b>&nbsp;')  # fancy separator
 
136     end
