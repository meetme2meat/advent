46   class ColorScheme 
47     #
 
48     # Create an instance of HighLine::ColorScheme. The customization can
 
49     # happen as a passed in Hash or via the yielded block.  Key's are
 
50     # converted to <tt>:symbols</tt> and values are converted to HighLine
 
51     # constants.
 
52     #
 
53     def initialize( h = nil )
 
54       @scheme = Hash.new
 
55       load_from_hash(h) unless h.nil?
 
56       yield self if block_given?
 
57     end

59     # Load multiple colors from key/value pairs.
 
60     def load_from_hash( h )
 
61       h.each_pair do |color_tag, constants|
 
62         self[color_tag] = constants
 
63       end
 
64     end

66     # Does this color scheme include the given tag name?
 
67     def include?( color_tag )

69     end
